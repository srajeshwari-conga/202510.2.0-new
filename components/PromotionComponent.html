projects/congarevenuecloud/elements/src/lib/promotion/promotion.component.ts

Description

Promotion component is used to apply and remove promotions on the cart. Also fetches the list of adjustment line item associated with the cart.

Preview

Usage

import { PromotionModule } from '@congarevenuecloud/elements';

@NgModule({
imports: [PromotionModule, ...]
})
export class AppModule {}
