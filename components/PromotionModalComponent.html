Description

Promotion modal component is used to show a modal window with all the details of promotions that are applied to items in the cart.

Preview

Usage

import { PromotionModalModule } from '@congarevenuecloud/elements';

@NgModule({
imports: [PromotionModalModule, ...]
})
export class AppModule {}

  <div class="modal-header d-flex flex-row-reverse px-0 pb-1">

      <span aria-hidden="true">&times;</span>
    </button>
  </div>
  <ng-container *ngIf="record$ | async as record; else loadingA">
    <div class="modal-body bg-white small pb-4">
      <div *ngIf="showApplyPromotion">

          <h5 class="modal-title">{{'PROMOTION.PROMOTIONS' | translate}}</h5>
        </div>
        <h5>{{'PROMOTION.APPLY_PROMOTION' | translate}}</h5>
        <form [ngClass]="adjustmentLIs?.length > 0 ? '':'pb-2'">
          <apt-input-field [entity]="incentive" [field]="'Name'" [fieldType]="'multipicklist'"
            [displayFieldType]="'dropdown'" [showLabel]="false" [(ngModel)]="promoCode" [readonly]="readOnly"
            [ngModelOptions]="{standalone: true}" [picklistValues]="picklistValues"
            (focusout)="updatePromotion(promoCode)"></apt-input-field>
        </form>

        <div *ngIf="availablePromotions?.length > 0">
          <h5>{{'PROMOTION.AVAILABLE_PROMOTIONS' | translate}}</h5>
        </div>

        <ul class="list-group mt-2 overflow-auto">
          <li class="list-group-item dashed-border border-secondary bg-white d-flex justify-content-between pb-2 mb-3"
             *ngFor="let promotion of availablePromotions">
            <div class="w-100">
              <div class="d-flex align-items-center justify-content-between mr-0">
                <div class="font-weight-bold">{{promotion?.IncentiveName}}</div>
                <span class="badge badge-success ml-2">{{promotion?.IncentiveCode}}</span>
              </div>
              <p [innerHTML]="promotion?.Description" class="mb-1"></p>

              <small class="text-muted">
                  {{promotion?.EndDate | date:'mediumDate'}} 
              </small>
              <div class="d-flex align-items-center w-100 justify-content-between">
                <span class="text-success font-weight-bold">{{promotion?.IncentiveAdjustmentAmount | localCurrency | async}}</span>
                <button class="btn btn-sm btn-outline-primary" (click)="updatePromotion([promotion?.IncentiveCode])">{{'COMMON.APPLY' | translate}}</button>
              </div>
            </div>
          </li>
        </ul>
      </div>
      <div *ngIf="adjustmentLIs?.length > 0">
        <h5 class="pt-3 pb-1">{{'PROMOTION.APPLIED_PROMOTIONS' | translate}}</h5>
      </div>
      <span *ngIf="adjustmentLIs?.length >0">
        {{'PROMOTION.PROMOTION_APPLIED' | translate}} <strong [innerHTML]="record?.Product?.Name"></strong>
      </span>
      <ul class="list-group mt-2 overflow-auto" *ngIf="adjustmentLIs?.length > 0">
        <li class="list-group-item dashed-border border-secondary bg-light d-flex justify-content-between mb-3 pb-2"
          *ngFor="let adjustment of adjustmentLIs">
          <div class="w-100">
            <div class="d-flex align-items-center justify-content-between mr-0">
              <div class="font-weight-bold">{{adjustment?.Incentive?.Name}}</div>
              <span class="badge badge-success ml-2">{{adjustment.CouponCode ? adjustment.CouponCode :
                adjustment?.IncentiveCode}}</span>
            </div>
            <p [innerHTML]="adjustment?.Incentive?.Description" class="mb-1"></p>
            <small class="text-muted">
              {{ adjustment?.Incentive?.ExpirationDate | date: 'mediumDate' }}
              <span class="mx-2">|</span>
              {{ adjustment?.LineType }}
            </small>
            <div class="d-flex align-items-center w-100 justify-content-between">
              <span class="text-success font-weight-bold">{{adjustment?.IncentiveAdjustmentAmount | localCurrency | async}}</span>
              <button *ngIf="!adjustment?.AutoApply && showApplyPromotion" class="btn btn-sm btn-outline-danger" (click)="updatePromotion()">{{'COMMON.REMOVE' | translate}}</button>
            </div>
          </div>
        </li>
      </ul>
    </div>


  </ng-container>
  <div class="modal-footer bg-white d-flex justify-content-between font-weight-bold border-top" *ngIf="adjustmentLIs?.length >0">

  </div>
